import base64

#to serialise
with open("oldmap.pgm", "rb") as imageFile:
    str = base64.b64encode(imageFile.read())
    print str

# to deserialse 
fh = open("deserialised.pgm", "wb")
fh.write(str.decode('base64'))
fh.close()
